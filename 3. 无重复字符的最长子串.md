class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int ans=0,flag=0;
        map<char,int> mp;
        for(int i=0,j=0;i<s.size();i++)
        {
            mp[s[i]]++;
            if(mp[s[i]]<=1) flag++;
            else
            {
                while(mp[s[i]]>=2)
                {
                    mp[s[j]]--;
                    j++;
                }
            }
            ans=max(ans,i-j+1);
        }
        return ans;
    }
};
