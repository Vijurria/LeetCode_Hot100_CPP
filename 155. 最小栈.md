class MinStack {
public:
    stack<int> stk,min_stack;
    MinStack() {
        
    }
    
    void push(int val) {
        stk.push(val);
        if(min_stack.size()==0||min_stack.top()>=val)
        {
            min_stack.push(val);
        }
    }
    
    void pop() {
        if(min_stack.top()==stk.top()) min_stack.pop();
        stk.pop();
    }
    
    int top() {
        return stk.top();
    }
    
    int getMin() {
        return min_stack.top();
    }
};
