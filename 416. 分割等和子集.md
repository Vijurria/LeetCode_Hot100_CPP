class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int n=nums.size(),sum=0;
        for(int i=0;i<n;i++)
        {
            sum+=nums[i];
        }
        if(sum%2!=0) return false;
        int flag=sum/2;
        vector<bool> f(flag+100);
        f[0]=true;
        for(int i=0;i<n;i++)
        {
            int x=nums[i];
            for(int j=flag;j>=x;j--)
            {
                f[j]=f[j]||f[j-x];
            }
        }
        return f[flag];
    }
};
