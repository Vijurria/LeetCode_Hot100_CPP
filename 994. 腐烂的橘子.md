class Solution {
public:
    typedef pair<int,int> PII;
    queue<PII> q;
    int dx[4]={-1,0,0,1},dy[4]={0,1,-1,0};
    int st[15][15];
    int time=0;
    void bfs(vector<vector<int>>& grid)
    {
        int n=grid.size(),m=grid[0].size();
        while(q.size())
        {
            auto t=q.front();
            q.pop();
            for(int i=0;i<4;i++)
            {
                int xx=t.first+dx[i],yy=t.second+dy[i];
                if(xx>=0&&xx<n&&yy>=0&&yy<m&&grid[xx][yy]==1&&st[xx][yy]==0)
                {
                    st[xx][yy]=st[t.first][t.second]+1;
                    time=max(time,st[xx][yy]);
                    q.push({xx,yy});
                }
            }
        }
    }
    int orangesRotting(vector<vector<int>>& grid) {
        int n=grid.size(),m=grid[0].size();
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(grid[i][j]==2)
                {
                    st[i][j]=1;
                    q.push({i,j});
                }
            }
        }
        bfs(grid);
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(st[i][j]==0&&grid[i][j]==1) return -1;
            }
        }
        return time==0?time:time-1;;
    }
};
