class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int zero=0,index=0;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]!=0) nums[index++]=nums[i];
            else zero++;
        }
        for(int i=index;i<nums.size();i++)
        {
            nums[i]=0;
        }
    }
};
