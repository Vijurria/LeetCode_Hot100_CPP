class Solution {
public:
    stack<string> stk;
    stack<int> num_stk;
    string ans,res;
    string decodeString(string s) {
        for(int i=0;i<s.size(); )
        {
            if(s[i]-'0'>=0&&s[i]-'0'<=9)
            {
                int num=0;
                while(s[i]-'0'>=0&&s[i]-'0'<=9)
                {
                    num=num*10+(s[i]-'0');
                    i++;
                }
                num_stk.push(num);
            }
            else if(s[i]>='a'&&s[i]<='z')
            {
                while(s[i]>='a'&&s[i]<='z')
                {
                    ans+=s[i];
                    i++;
                }
            }
            else if(s[i]=='[')
            {
                stk.push(ans);
                ans="";
                i++;
            }
            else if(s[i]==']')
            {
                string res=stk.top(); 
                stk.pop();
                int num=num_stk.top();
                num_stk.pop();
                for(int j=0;j<num;j++)
                {
                    res+=ans;
                }
                ans=res;
                i++;
            }
        }
        return ans;
    }
};
