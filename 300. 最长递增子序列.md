class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n=nums.size();
        vector<int> f(n,1);
        int maxn=1;
        for(int i=1;i<n;i++)
        {
            for(int j=0;j<i;j++)
            {
                if(nums[i]>nums[j])
                    f[i]=max(f[i],f[j]+1);
            }
            maxn=max(maxn,f[i]);
        }
        return maxn;
    }
};
