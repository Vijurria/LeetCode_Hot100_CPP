class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int n=nums.size();
        vector<int> ans(n);
        for(int i=0;i<nums.size();i++)
        {
            ans[i]=nums[i];
        }
        for(int i=0;i<ans.size();i++)
        {
            nums[(i+k)%n]=ans[i];
        }
    }
};
