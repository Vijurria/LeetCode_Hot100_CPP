class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int ans=1;
        for(int i=0;i<nums.size();i++)
        {
            if(i>=1&&nums[i]==nums[i-1]) continue;
            if(nums[i]>=1)
            {
                if(nums[i]==ans) ans++;
                else break;
            }
        }
        return ans;
    }
};
