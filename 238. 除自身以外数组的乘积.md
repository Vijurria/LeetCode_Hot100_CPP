class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int n=nums.size();
        vector<int> ans(n);
        int sum=1,zero=0;
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]==0) zero++;
            else sum*=nums[i];
        }
        for(int i=0;i<nums.size();i++)
        {
            if(nums[i]==0)
            {
                if(zero>=2) ans[i]=0;
                else ans[i]=sum;
            }
            else
            {
                if(zero>=1) ans[i]=0;
                else ans[i]=sum/nums[i];
            }
        }
        return ans;
    }
};
