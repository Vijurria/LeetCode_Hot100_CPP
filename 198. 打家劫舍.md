class Solution {
public:
    int rob(vector<int>& nums) {
        vector<int> f(nums);
        for(int i=0;i<nums.size();i++)
        {
            f[i]=nums[i];//初始时只偷一个人家
        }
        int maxn=0;
        for(int i=0;i<nums.size();i++)
        {
            for(int j=0;j<i-1;j++)
            {
                f[i]=max(f[i],f[j]+nums[i]);//从前往后看目前更大还是这一家+别家大
            }
            maxn=max(maxn,f[i]);
        }
        return maxn;
    }
};
