class Solution {
public:
    vector<vector<int>> ans;
    vector<int> res;
    bool flag[10];
    void dfs(vector<int>& nums)
    {
        if(res.size()==nums.size())
        {
            ans.push_back(res);
            return ;
        }
        for(int i=0;i<nums.size();i++)
        {
            if(flag[i]==false)
            {
                res.push_back(nums[i]);
                flag[i]=true;
                dfs(nums);
                res.pop_back();
                flag[i]=false;
            }
        }
    }
    vector<vector<int>> permute(vector<int>& nums) {
        dfs(nums);
        return ans;
    }
};
