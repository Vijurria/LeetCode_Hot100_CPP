class Solution {
public:
    vector<vector<string>> ans;
    vector<string> res;
    bool hw(string s)
    {
        if(s.size()==1) return true;
        for(int i=0;i<s.size()/2+1;i++)
        {
            if(s[i]!=s[s.size()-1-i]) 
                return false;
        }
        return true;
    }
    void dfs(int idx,string s)
    {
        if(idx==s.size())//已处理完整个字符串
        {
            ans.push_back(res);
            return ;
        }
        for(int i=idx;i<s.size();i++)//尝试所有可能的分割点
        {
            if(hw(s.substr(idx,i-idx+1)))//当前子串 s[idx..i]
            {
                res.push_back(s.substr(idx,i-idx+1));
                dfs(i+1,s);
                res.pop_back();
            }
        }
    }
    vector<vector<string>> partition(string s) {
        dfs(0,s);//从字符串起始位置（0）开始递归
        return ans;
    }
};
