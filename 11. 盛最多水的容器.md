class Solution {
public:
    int maxArea(vector<int>& height) {
        int l=0,r=height.size()-1;
        int maxn=0;
        while(l<r)
        {
            maxn=max(maxn,min(height[l],height[r])*(r-l));
            if(height[l]<=height[r]) l++;
            else r--;
        }
        return maxn;
    }
};
