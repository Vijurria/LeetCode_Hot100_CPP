class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> ans;
        vector<int> v;
        for(int i=1;i<=numRows;i++)
        {
            for(int j=1;j<=i;j++)
            {
                if(j==1||j==i) v.push_back(1);
                else
                {
                    int r=ans.size();
                    int sum=ans[r-1][j-2]+ans[r-1][j-1];
                    v.push_back(sum);
                }
            }
            ans.push_back(v);
            v.clear();
        }
        return ans;
    }
};
