class Solution {
public:
    int findDuplicate(vector<int>& nums) {
        int fast=0,low=0;
        do
        {
            low=nums[low];
            fast=nums[nums[fast]];
        }while(fast!=low); //首先使用快慢指针找到环结尾的点，也就是即将相遇的点
        low=0;
        while(fast!=low)//快指针重复这个环，等慢指针相遇
        {
            low=nums[low];
            fast=nums[fast];
        }
        return fast;
    }
};
