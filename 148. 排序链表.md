/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        vector<int> v;
        ListNode* beg=head;
        while(beg)
        {
            v.push_back(beg->val);
            beg=beg->next;
        }
        sort(v.begin(),v.end());
        beg=head;
        int i=0;
        while(beg)
        {
            beg->val=v[i++];
            beg=beg->next;
        }
        return head;
    }
};
