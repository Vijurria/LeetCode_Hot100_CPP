class Solution {
public:
    vector<string> ans;
    void dfs(int l,int r,int& n,string& s)
    {
        if(l+r==2*n)
        {
            ans.push_back(s);
            return ;
        }
        if(l<n)
        {
            s+="(";
            dfs(l+1,r,n,s);
            s.pop_back();
        }
        if(r<n&&r<l)
        {
            s+=")";
            dfs(l,r+1,n,s);
            s.pop_back();
        }
    }
    vector<string> generateParenthesis(int n) {
        string s="(";
        dfs(1,0,n,s);
        return ans;
    }
};
