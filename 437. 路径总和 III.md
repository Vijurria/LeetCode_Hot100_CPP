/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    long long int target,ans;
    void dfs2(TreeNode *node,long long int sum)
    {
        sum+=node->val;
        if(sum==target) ans++;
        if(node->left!=nullptr) dfs2(node->left,sum);
        if(node->right!=nullptr) dfs2(node->right,sum);
    }
    void dfs(TreeNode* node)
    {
        dfs2(node,0);
        if(node->left!=nullptr) dfs(node->left);
        if(node->right!=nullptr) dfs(node->right);
    }
    int pathSum(TreeNode* root,long long int targetSum) {
        target=targetSum;
        if(root==nullptr) return 0;
        dfs(root);
        return ans;
    }
};
