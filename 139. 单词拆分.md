class Solution {
public:
    bool wordBreak(string s, vector<string>& wordDict) {
        int n=s.size();
        vector<bool> f(n,false);
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<wordDict.size();j++)
            {
                int len=wordDict[j].size();
                if(i+1<len) continue; //如果当前s的长度不超过word长度
                if(i+1==len) //刚好从0开始那么长
                {   //判断字符串的开头部分是否能匹配字典中的某个单词
                    f[i]=f[i]||(s.substr(0,len)==wordDict[j]);
                }
                else
                {   //确保当前单词前面的部分也能被成功拆分
                    f[i]=f[i]||(f[i-len]&&(s.substr(i-len+1,len)==wordDict[j]));
                }
            }
        }
        return f[n-1];
    }
};
