class Solution {
public:
    bool canFinish(int numCourses, vector<vector<int>>& prerequisites) {
        vector<vector<int>> ans(numCourses,vector<int>());
        vector<int> in(numCourses);
        for(int i=0;i<prerequisites.size();i++)
        {
            ans[prerequisites[i][1]].push_back(prerequisites[i][0]);//形成树结构
            in[prerequisites[i][0]]++;//入度++
        }
        queue<int> q;
        for(int i=0;i<numCourses;i++)
        {
            if(in[i]==0) q.push(i);//没有入度的放进去
        }
        int sum=0;
        while(q.size())
        {
            int t=q.front();
            q.pop();
            sum++;// 学完一门课
            for(int i=0;i<ans[t].size();i++)
            {
                in[ans[t][i]]--; // 学完 t，t 指向的课程的先修需求减 1
                if(in[ans[t][i]]==0) q.push(ans[t][i]);// 如果某门课所有先修都学完了，入队
            }
        }
        return sum==numCourses;
    }
};
