class Solution {
public:
    vector<vector<string>> ans; //存储所有解
    int col[20],dg[20],udg[20];//对角线、反对角线标记
    void dfs(int n,int u,vector<string>& g)
    {
        if(u==n) //找到一个解，保存当前棋盘状态
        {
            ans.push_back(g);
            return ;
        }
        for(int i=0;i<n;i++)
        {
            if(col[i]==0&&dg[u+i]==0&&udg[n-u+i]==0)
            {
                g[u][i]='Q'; //放皇后
                col[i]=dg[u+i]=udg[n-u+i]=true; //标记
                dfs(n,u+1,g); //递归下一行
                col[i]=dg[u+i]=udg[n-u+i]=false; //恢复标记
                g[u][i]='.';//回溯
            }
        }
    }
    vector<vector<string>> solveNQueens(int n) {
        vector<string> g(n,string(n,'.'));
        dfs(n,0,g);
        return ans;
    }
};
