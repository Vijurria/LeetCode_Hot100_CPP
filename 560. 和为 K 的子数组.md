class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int n=nums.size();
        int a[n];
        for(int i=0;i<n;i++)
        {
            a[i]=nums[i];
        }
        int sum=0;
        for(int i=0;i<n;i++)
        {
            int flag=0;
            for(int j=i;j>=0;j--)
            {
                flag+=a[j];
                if(flag==k) sum++;
            }
        }
        return sum;
    }
};
