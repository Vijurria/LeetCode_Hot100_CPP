class Solution {
public:
    stack<int> stk;
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n=temperatures.size();
        vector<int> ans(n,0);
        for(int i=0;i<temperatures.size();i++)
        {
            while(stk.size()&&temperatures[i]>temperatures[stk.top()])
            {
                int idx=stk.top();
                stk.pop();
                ans[idx]=i-idx;
            }
            stk.push(i);
        }
        return ans;
    }
};
