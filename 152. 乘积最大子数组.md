class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int n=nums.size();
        vector<int> zs(n),fs(n);
        zs[0]=fs[0]=nums[0];
        int maxn=nums[0];
        for(int i=1;i<n;i++)
        {
            zs[i]=max({nums[i],zs[i-1]*nums[i],fs[i-1]*nums[i]});
            fs[i]=min({nums[i],zs[i-1]*nums[i],fs[i-1]*nums[i]});
            maxn=max({maxn,zs[i],fs[i]});
        }
        return maxn;
    }
};
