class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        map<char,int> mp,mps;
        for(int i=0;i<p.size();i++)
        {
            mp[p[i]]++;
        }
        vector<int> ans;
        for(int i=0,j=0;i<s.size();i++)
        {
            mps[s[i]]++;
            while(i-j+1>p.size()) mps[s[j]]--,j++;
            if(i-j+1==p.size())
            {
                bool flag=true;
                for(int aa='a';aa<='z';aa++)
                {
                    if(mp[char(aa)]!=mps[char(aa)])
                    {
                        flag=false;
                        break;
                    }
                }
                if(flag==true) ans.push_back(j);
            }
        }
        return ans;
    }
};
