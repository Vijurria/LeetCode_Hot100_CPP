class Solution {
public:
    int dx[4]={-1,0,0,1},dy[4]={0,1,-1,0};
    bool vis[100][100];
    bool flag=false;
    string s;
    void dfs(int x,int y,int idx,vector<vector<char>>& board, string& word)
    {
        if(idx==word.size())
        {
            flag=true;
            return ;
        }
        int n=board.size(),m=board[0].size();
        for(int i=0;i<4;i++)
        {
            int xx=x+dx[i],yy=y+dy[i];
            if(xx>=0&&xx<n&&yy>=0&&yy<m&&board[xx][yy]==word[idx]&&vis[xx][yy]==false)
            {
                vis[x][y]=true;
                dfs(xx,yy,idx+1,board,word);
                vis[x][y]=false;
            }
        }
    }
    bool exist(vector<vector<char>>& board, string word) {
        memset(vis,0,sizeof vis);
        int n=board.size(),m=board[0].size();
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(board[i][j]==word[0])
                {
                    vis[i][j]=true;
                    dfs(i,j,1,board,word);
                    vis[i][j]=false;
                }
                if(flag==true) return true;
            }
        }
        return flag;
    }
};
