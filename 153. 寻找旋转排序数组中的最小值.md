class Solution {
public:
    int findMin(vector<int>& nums) {
        int l=0,r=nums.size()-1;
        while(l<r)
        {
            int mid=(l+r)/2;
            if(nums[l]<=nums[mid])//nums l比mid小 1 2 3     8 9 1
            {
                if(nums[mid]<nums[r]) r=mid;// 1 2 3
                else l=mid+1;//8 9 1
            }
            else//nums l比mid大 9 2 8 
            {
                r=mid;
            }
        }
        return nums[r];
    }
};
