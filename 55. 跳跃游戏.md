class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n=nums.size(),maxn=0;
        vector<int> v(n+1,0);
        for(int i=0;i<nums.size()-1;i++)
        {
            if(maxn>=i&&nums[i]!=0) maxn=max(maxn,i+nums[i]);
        }
        return maxn+1>=n?true:false;
    }
};
