class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        vector<int> ans;
        for(int i=0;i<nums.size();i++)
        {
            if(i==0||nums[i]!=nums[i-1]) ans.push_back(nums[i]);
        }
        int maxn=0,flag=0;
        for(int i=0;i<ans.size();i++)
        {
            if(i==0||ans[i]==ans[i-1]+1)
            {
                flag++;
                maxn=max(maxn,flag);
            }
            else flag=1;
        }
        return maxn;
    }
};
