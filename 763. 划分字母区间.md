class Solution {
public:
    vector<int> partitionLabels(string s) {
        s=" "+s;
        map<char,pair<int,int>> mp;
        for(int i=1;i<s.size();i++)
        {
            if(mp[s[i]].first==0) mp[s[i]].first=i,mp[s[i]].second=i;
            else mp[s[i]].second=i;
        }
        vector<int> ans;
        int l=0,r=0;
        for(int i=1;i<s.size();i++)
        {
            if(i==1)
            {
                l=mp[s[i]].first;
                r=mp[s[i]].second;
            }
            else if(mp[s[i]].first<r) r=max(r,mp[s[i]].second);
            else if(mp[s[i]].first>=r)
            {
                ans.push_back(r-l+1);
                l=mp[s[i]].first;
                r=mp[s[i]].second;
            }
        }
        ans.push_back(r-l+1);
        return ans;
    }
};
