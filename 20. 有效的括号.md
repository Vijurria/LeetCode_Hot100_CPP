class Solution {
public:
    stack<char> stk;
    bool isValid(string s) {
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='(') stk.push('(');
            else if(s[i]=='[') stk.push('[');
            else if(s[i]=='{') stk.push('{');
            else if(s[i]==')'&&stk.size()&&stk.top()=='(') stk.pop();
            else if(s[i]==']'&&stk.size()&&stk.top()=='[') stk.pop();
            else if(s[i]=='}'&&stk.size()&&stk.top()=='{') stk.pop();
            else return false;
        }
        if(stk.size()) return false;
        else return true;
    }
};
