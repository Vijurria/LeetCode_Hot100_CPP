class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        map<int,int> mp;
        int max_cnt=0;
        for(int i=0;i<nums.size();i++)
        {
            mp[nums[i]]++;
            max_cnt=max(max_cnt,mp[nums[i]]);
        }
        vector<vector<int>> v(max_cnt+1);
        for(auto& [x,y]:mp)
        {
            v[y].push_back(x);//把有多少次数下的数字放入边长数组中
        }
        vector<int> ans;
        for(int i=max_cnt; ;i--)
        {
            if(ans.size()<k)
            {
                ans.insert(ans.end(),v[i].begin(),v[i].end());//在答案中依次插入数字
            }
            else break;
        }
        return ans;
    }
};
