class Solution {
public:
    vector<string> mp={" "," ","abc","def","ghi","jkl","mno","pqrs","tuv","wxyz"};
    vector<string> ans;
    string flag;
    void dfs(int idx,string digits)
    {
        if(idx==digits.size())
        {
            ans.push_back(flag);
            return ;
        }
        int num=digits[idx]-'0';
        for(int i=0;i<mp[num].size();i++)
        {
            flag+=mp[num][i];
            dfs(idx+1,digits);
            flag.pop_back();
        }
    }
    vector<string> letterCombinations(string digits) {
        dfs(0,digits);
        return ans;
    }
};
