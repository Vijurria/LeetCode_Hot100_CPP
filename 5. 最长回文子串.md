class Solution {
public:
    string longestPalindrome(string s) {
        int n=s.size();
        int al=0,ar=0;
        for(int i=0;i<n;i++)
        {
            int l=i,r=i;
            while(l>=0&&r<n&&s[l]==s[r])
            {
                l--; r++;
            }
            if(r-l-1>ar-al+1)
            {
                al=l+1; ar=r-1;
            }
        }
        for(int i=0;i<n-1;i++)
        {
            int l=i,r=i+1;
            while(l>=0&&r<n&&s[l]==s[r])
            {
                l--; r++;
            }
            if(r-l-1>ar-al+1)
            {
                al=l+1; ar=r-1;
            }
        }
        string ans=s.substr(al,ar-al+1);
        return ans;
    }
};
